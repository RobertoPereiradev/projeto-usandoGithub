vamos ver!

git status //verifica atualizações
git init // inicializa um ovo repositório
git add "[(nome)]" //adiciona um arquivo ou sua atualização
git add . //adiciona um arquivo
git commit -m "[(nome)]" //cria e dá nome à nova versão do arquivo
git remote add origin [(link do site)] //localização da nuvem
git push //coloca na nuvem
git reflog //histórico, com a mais atual no topo
git reset --hard [(id)] //mudar entre as versões do branch

git branch // mostra a branch principal
git branch [(nome)] // o nome comum é staging; adiciona outra branch(galho)
git checkout [(nome)] // troca para a branch em questão
git merge [(nome)] // atualiza a branch em questão com a atual
git pull // atualiza a brach atual com os dados mais novos da nuvem (usado em desenvolvimentos mutuos)
git checkout -b [(nome da branch)] [(nome da branch em que será copiada)] // cria uma branch a partir de uma cópia de outra branch

touch .gitignore // cria um arquivo para colocar o caminho completo do que deve ser ignorado (deve ser adicionado depois)
pull request // é feito no github. Nele você escolhe de onde sai e de onde vai esse pull
